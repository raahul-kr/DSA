class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        // Step 1: Bundle each number with its original index
        vector<pair<int, int>> arr;
        for (int i = 0; i < nums.size(); i++) {
            arr.push_back({nums[i], i});
        }
        
        // Step 2: Sort the bundles (sorts automatically by the numbers)
        sort(arr.begin(), arr.end());
        
        // Step 3: The Easy Two-Pointer Logic
        int left = 0;
        int right = arr.size() - 1;
        
        while (left < right) {
            int sum = arr[left].first + arr[right].first;
            
            if (sum == target) {
                // Bingo! Return the original indices we bundled earlier
                return {arr[left].second, arr[right].second};
            } 
            else if (sum < target) {
                // Sum is too small? Move the left pointer up to get a bigger number
                left++;
            } 
            else {
                // Sum is too big? Move the right pointer down to get a smaller number
                right--;
            }
        }
        
        return {};
    }
};
