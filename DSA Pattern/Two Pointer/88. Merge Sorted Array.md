class Solution {
public:
    void merge(vector<int>& nums1, int m, vector<int>& nums2, int n) {
        vector<int>& a = nums1;
        vector<int>& b = nums2;

        int i = 0;
        int j = 0;
        
        vector<int> res(m + n);
        int id = 0;

        while (i < m && j < n) {
            if (a[i] <= b[j]) {
                res[id] = a[i];
                id++;
                i++;
            } else {
                res[id] = b[j];
                id++;
                j++;
            }
        }

        while (j < n) {
            res[id] = b[j];
            id++;
            j++;
        }
        while (i < m) {
            res[id] = a[i];
            id++;
            i++; 
        }

        nums1 = res;
    }
};
