class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        if (nums.empty()) return 0;
        
        // 'i' is the slow pointer tracking the last unique element
        int i = 0;
        
        // 'j' is the fast pointer scanning for new unique elements
        for (int j = 1; j < nums.size(); j++) {
            if (nums[j] != nums[i]) {
                i++; // Move the slow pointer forward
                nums[i] = nums[j]; // Update the position with the new unique value
            }
        }
        
        // The number of unique elements is the index + 1
        return i + 1;
    }
};
