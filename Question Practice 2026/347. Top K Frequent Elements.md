```cpp
#include <bits/stdc++.h>
using namespace std;

vector<int> topKFrequent(vector<int>& nums, int k) {
    unordered_map<int, int> freq;

    // count frequency
    for (int num : nums) {
        freq[num]++;
    }

    // move to vector
    vector<pair<int,int>> vec;
    for (auto it : freq) {
        vec.push_back({it.second, it.first}); // {freq, num}
    }

    // sort descending
    sort(vec.rbegin(), vec.rend());

    vector<int> result;
    for (int i = 0; i < k; i++) {
        result.push_back(vec[i].second);
    }

    return result;
}

```

### To solve this question i can  use the hashmap to count the freq of each unique number and Bucket Sort strategy by creating a List of Lists where each index represents a frequency, placing numbers into the bucket that matches their count. Finally, you scan this list from the highest index to the lowest, collecting numbers until you reach k elements, which achieves an efficient $O(n)$ linear time complexity

#include <vector>
#include <unordered_map>

using namespace std;

vector<int> topKFrequent(vector<int>& nums, int k) {
    // 1. Map to store: {number -> frequency}
    unordered_map<int, int> counts;
    for (int n : nums) {
        counts[n]++;
    }

    // 2. A list where the index is the frequency. 
    // We use a vector of vectors because multiple numbers can have the same frequency.
    vector<vector<int>> buckets(nums.size() + 1);
    for (auto const& [num, freq] : counts) {
        buckets[freq].push_back(num);
    }

    // 3. Iterate backwards from the highest possible frequency
    vector<int> result;
    for (int i = buckets.size() - 1; i >= 0; i--) {
        for (int num : buckets[i]) {
            result.push_back(num);
            if (result.size() == k) return result;
        }
    }
    return result;
}
