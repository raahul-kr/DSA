#include <bits/stdc++.h>
using namespace std;

class Solution {
public:
    bool isAnagram(string s, string t) {
        if (s.size() != t.size()) return false;

        unordered_map<char,int> countS, countT;

        for (int i = 0; i < s.length(); i++) {
        countS[s[i]]++;
        countT[t[i]]++;
        }

        return countS == countT;
    }
};
